pipeline {
    agent any
    
    stages {
        stage('Build') {
            steps {
                echo 'Building the project...'
                // Example: sh 'mvn clean package'
            }
        }
        stage('Unit and Integration Tests') {
            steps {
                echo 'Running unit and integration tests...'
                // Example: sh 'mvn test'
            }
        }
        stage('Code Analysis') {
            steps {
                echo 'Performing code analysis...'
                // Example: sh 'sonar-scanner'
            }
        }
        stage('Security Scan') {
            steps {
                echo 'Running security scan...'
                // Example: sh 'npm audit'
            }
        }
        stage('Deploy to Staging') {
            steps {
                echo 'Deploying to staging server...'
                // Example: sh 'scp target/app.jar user@staging-server:/path/to/deploy'
            }
        }
        stage('Integration Tests on Staging') {
            steps {
                echo 'Running integration tests on staging...'
                // Example: sh 'curl http://staging-server/health-check'
            }
        }
        stage('Deploy to Production') {
            steps {
                echo 'Deploying to production server...'
                // Example: sh 'scp target/app.jar user@production-server:/path/to/deploy'
            }
        }
    }
    
    post {
        always {
            // Archive the console log to make it accessible for reading
            archiveArtifacts artifacts: '**/target/*.log', allowEmptyArchive: true
            
            // Read the Jenkins console log
            script {
                def logContent = currentBuild.rawBuild.getLog(100).join('\n')
                def logFile = writeFile(file: 'pipeline-log.txt', text: logContent)
            }
        }

        success {
            mail to: 'dasankalana3@gmail.com',
                 subject: "Pipeline Successful: ${currentBuild.fullDisplayName}",
                 body: "The Jenkins pipeline has completed successfully.\n\nLogs:\n" + logContent
        }
        failure {
            mail to: 'dasankalana3@gmail.com',
                 subject: "Pipeline Failed: ${currentBuild.fullDisplayName}",
                 body: "The Jenkins pipeline has failed. Please check the logs for details.\n\nLogs:\n" + logContent
        }
    }
}
